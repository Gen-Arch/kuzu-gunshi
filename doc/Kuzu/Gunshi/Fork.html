def form(num)
  childs = Array.new
  num.times do |i|
    klass = Kuzu::Gunshi.const_set :"ChildGunshi#{i}", Class.new(Template::ForkTemp)
    if rand(10) == 7
      klass.class_eval{ def you_are_name?; kill end }
    end
    (childs ||= "") << klass
  end
  return childs
end
